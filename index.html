import random class Clique: def __init__(self, name): self.name = name self.power = 10 self.territory_count = 1 self.funds = 100 def __str__(self): return f"{self.name} | Power: {self.power} | Territory: {self.territory_count} | Funds: ${self.funds}" class Sector: def __init__(self, name, difficulty): self.name = name self.difficulty = difficulty self.owner = "Neutral" def complete_mission(clique): print(f"\n--- Mission Available: Training Camp ---") success_chance = random.randint(1, 10) if success_chance > 3: reward = 50 power_gain = 5 clique.funds += reward clique.power += power_gain print(f"Success! You gained ${reward} and {power_gain} power.") else: print("Mission failed. Your fighters need more rest.") def take_over_sector(clique, sector): print(f"\n--- Attempting to take over {sector.name} ---") if clique.power >= sector.difficulty: sector.owner = clique.name clique.territory_count += 1 clique.power -= 5 # Cost of expansion print(f"Victory! {sector.name} is now under your control.") else: print(f"Defeat! {sector.name} was too strong. Increase your power first.") # --- Game Setup --- player_clique = Clique("The Iron Fists") map_sectors = [ Sector("The Docks", 12), Sector("Downtown Gym", 25), Sector("The Industrial Hub", 40) ] # --- Mini Game Loop --- while True: print("\n" + "="*30) print(player_clique) print("1. Do Mission 2. Take Territory 3. Exit") choice = input("Select an action: ") if choice == "1": complete_mission(player_clique) elif choice == "2": for i, s in enumerate(map_sectors): status = "[OWNED]" if s.owner == player_clique.name else f"[Diff: {s.difficulty}]" print(f"{i+1}. {s.name} {status}") target = int(input("Choose a sector: ")) - 1 take_over_sector(player_clique, map_sectors[target]) elif choice == "3": break
